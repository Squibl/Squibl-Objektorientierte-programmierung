#include <iostream>
#include <string.h>
#include <vector>
#include <string>

using namespace std;

// String-Class
class String
{
    vector<char> str;

public:
    // Konstruktor
    String(){}
    String(const std::string strInput)
    {
        for (size_t i = 0; i < strInput.size(); i++)
        {
            str.push_back(strInput[i]);
        }
        
    }
    // Destructor memory release
    ~String(){}

    // Output Mehtod
    void print()
    {
        for (int i=0; i < str.size(); i++)
        {
            std::cout << str.at(i);
        }
            std::cout << "\n";

    }

    // String merge
    void add(const std::string strInput)
    { 
        for (size_t i = 0; i < strInput.size(); i++)
        {
            str.push_back(strInput[i]);
        }
    }

    // Size String
    void size()
    {
        std::cout << "String lenght:" << str.size() << "\n";
    }
    // String Test auf Gleichheit
    int isEqual(const std::string equalChar)
    {
        int vergleich=1;
        for (size_t i = 0; i < str.size(); i++)
        {
            if (str.at(i)!=equalChar.at(i))
            {
                vergleich=0;
                return vergleich;
            }
        }
        return vergleich;
    }

    int find(const std::string equalChar)
    {
        int inputGröße = equalChar.size();
        int strObjekt = str.size();

        // Wenn der zu suchende String länger ist als der Hauptstring, kann er nicht enthalten sein
        if (inputGröße > strObjekt)
        {
            return -1;
        }

        for (int i = 0; i <= strObjekt - inputGröße; i++)
        {
            int gefunden = true;
            for (int strIndex = 0; strIndex < inputGröße; strIndex++)
            {
                if (str.at(i + strIndex) != equalChar.at(strIndex))
                {
                    gefunden = false;
                    break;
                }
            }
            if (gefunden)
            {
                return i; // Start-Position zurückgeben
            }
        }
        return -1; // Nicht gefunden
    }
    void zuweisung(const String& s);
};

int main()
{
    String Name("FH:");
    Name.print();
    Name.size();

    Name.add("Emden/Leer");
    Name.print();
    Name.size();

    cout << Name.find("Emden");
    //cout << Name.find("mden");

    return 0;
}
