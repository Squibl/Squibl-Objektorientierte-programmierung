template<typename T>
void sortier(T const &start, T const &ende){
int n = (int)(ende - start) ;
bool swapped;
do{
    swapped = false;
for(int i = 0; i < n-1; i++)
{
    if(start[i]  > start[i+1]){
        swap(start[i],start[i+1]);
        swapped = true;
    }
}
n--;
}while (swapped);
}
